\begin{align} \mathbb{P}(X_4=3\mid X_2=1) &= \sum_{i=1}^3\mathbb{P}(X_4=3,X_3=i\mid X_2=1) \\ &= \sum_{i=1}^3\mathbb{P}(X_4=3\mid X_3=i,X_2=1)\mathbb{P}(X_3=i\mid X_2=1) \\ &= \sum_{i=1}^3\mathbb{P}(X_4=3\mid X_3=i)\mathbb{P}(X_3=i\mid X_2=1) \tag{Markov Property} \\ &= \sum_{i=1}^3P_{i,3}P_{1,i} \\ &= (P^2)_{1,3}\,. \end{align}
